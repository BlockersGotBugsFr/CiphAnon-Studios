Automated Task Runner

A simple Python script to automate repetitive file management tasks.

# Python Automated Task Runner import os import shutil source_dir = "C:/Users/user/Desktop/reports" target_dir = "C:/Users/user/Desktop/archives" def move_files(): for filename in os.listdir(source_dir): if filename.endswith(".txt"): shutil.move(os.path.join(source_dir, filename), target_dir) if __name__ == "__main__": move_files()

Server Log Analyzer

An application to analyze server logs and generate easy-to-read reports.

# Server Log Analyzer (pseudo-code) def analyze_logs(log_file): with open(log_file, 'r') as f: log_data = f.readlines() errors = [line for line in log_data if "ERROR" in line] warnings = [line for line in log_data if "WARNING" in line] print(f"Found {len(errors)} errors and {len(warnings)} warnings.")
